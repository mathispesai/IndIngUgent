\cos(x), \Bgtan(x)
\(m_{\text{aarde}}\)
\(\si{kg m/s^2}\)

\(V_{\ce{H_2O}}\)
\end{verbatim}
        \end{exampleblock}
        \column{.5\textwidth}
        Wiskunde cursief tenzij
        \begin{itemize}
            \item Meerdere letters\\
            niet \(cos x\) maar \(\cos x\)\\
            niet \(Bgtan x\) maar \(\Bgtan x\)
            \item Lopende tekst in formule\\
            niet \(m_{aarde}\) maar \(m_{\text{aarde}}\)
            \item Fysische eenheden\\
            niet \(kg m/s^2\) maar \(\si{kg m/s^2}\)
            \item Namen van atomen\\
            niet \(V_{H_2O}\) maar \(V_{\ce{H2O}}\)
        \end{itemize}
    \end{columns}
\end{frame}

\end{itemize}

\begin{itemize}

\item \(\norm{\vec{v}}\) \\

\begin{itemize}

Bestanden installeren (Mac OS X/Linux zie \texttt{readme.pdf})
\begin{itemize}
\item Kopieer de inhoud van de \texttt{texmf}-map naar\newline
\verb|c:\users\<user>\texmf\tex\latex\kulak|
\item Start het programma \texttt{MikTeX Settings} via\newline
\texttt{Start}, \texttt{All programs}, \texttt{MikTeX 2.9},\newline \texttt{Maintenance}, \texttt{Settings}
\item Klik op \texttt{Roots}, \texttt{Add...}, selecteer de map
\verb|c:\users\<username>\texmf|
\end{itemize}
\end{frame}

\verb|~/.config/texstudio/templates/user|

Template openen:
\begin{itemize}

\item Klik op \texttt{File}, \texttt{New From Template...}
